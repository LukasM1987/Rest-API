DROP PROCEDURE IF EXISTS UserName;
DROP FUNCTION IF EXISTS UserNotExist;
DELIMITER $$
CREATE FUNCTION UserNotExist() RETURNS VARCHAR(20) DETERMINISTIC
    BEGIN
        DECLARE result VARCHAR(20) DEFAULT 'User not exist!';
        RETURN result;
    END $$
CREATE PROCEDURE UserName(id INT)
BEGIN
    IF id <= 0 THEN
        SELECT UserNotExist();
    ELSEIF id = 1  THEN
        SELECT FIRSTNAME, LASTNAME FROM READERS WHERE READER_ID = 1;
    ELSEIF id = 2  THEN
        SELECT FIRSTNAME, LASTNAME FROM READERS WHERE READER_ID = 2;
    ELSEIF id = 3  THEN
        SELECT FIRSTNAME, LASTNAME FROM READERS WHERE READER_ID = 3;
    ELSEIF id = 4  THEN
        SELECT FIRSTNAME, LASTNAME FROM READERS WHERE READER_ID = 4;
    ELSEIF id = 5  THEN
        SELECT FIRSTNAME, LASTNAME FROM READERS WHERE READER_ID = 5;
    END IF;
END $$
DELIMITER ;
CALL UserName(3);
